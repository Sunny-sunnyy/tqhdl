import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# ==========================================
# KHỐI 1: DATA PROFILING & CLEANING (LÀM SẠCH)
# Mục đích: Đọc dữ liệu, kiểm tra các giá trị null và chuẩn hóa kiểu dữ liệu.
# ==========================================

# Đọc dữ liệu từ file CSV
sales_df = pd.read_csv('Regional Sales Dataset.xlsx - Sales Orders.csv')

# Ép kiểu dữ liệu thời gian cho cột OrderDate để dễ phân tích theo tháng/năm
sales_df['OrderDate'] = pd.to_datetime(sales_df['OrderDate'])

# Kiểm tra dữ liệu bị thiếu (Missing Values)
# Logic: Phải biết dữ liệu thiếu ở đâu để có phương án điền (fill) hoặc bỏ (drop)
missing_values = sales_df.isnull().sum() 


# ==========================================
# KHỐI 2: UNIVARIATE & BIVARIATE ANALYSIS (PHÂN TÍCH ĐƠN BIẾN & SONG BIẾN)
# Mục đích: Hiểu sự phân bổ của Doanh thu (Revenue) và Lợi nhuận (Profit).
# ==========================================

# Tính toán Doanh thu và Lợi nhuận (Logic: Doanh thu = Số lượng * Đơn giá)
sales_df['Revenue'] = sales_df['Order Quantity'] * sales_df['Unit Price']
sales_df['Profit'] = sales_df['Revenue'] - (sales_df['Total Unit Cost'] * sales_df['Order Quantity'])

# Phân tích xem kênh bán hàng nào mang lại Doanh thu lớn nhất
# Logic: Nhóm (groupby) theo Kênh (Channel) và tính tổng doanh thu
channel_revenue = sales_df.groupby('Channel')['Revenue'].sum()


# ==========================================
# KHỐI 3: TREND & SEASONALITY (XU HƯỚNG & MÙA VỤ)
# Mục đích: Phân tích biến động doanh thu theo từng tháng để lên kế hoạch tồn kho.
# ==========================================

# Trích xuất tháng và năm từ cột ngày tháng
sales_df['Month_Year'] = sales_df['OrderDate'].dt.to_period('M')

# Tính tổng doanh thu theo từng tháng
monthly_trend = sales_df.groupby('Month_Year')['Revenue'].sum()


# ==========================================
# KHỐI 4: CORRELATION & OUTLIER DETECTION (TƯƠNG QUAN & NGOẠI LỆ)
# Mục đích: Tìm ra các giao dịch bất thường và mối liên hệ giữa các chỉ số.
# ==========================================

# Tính toán ma trận tương quan để xem Đơn giá, Số lượng và Lợi nhuận liên quan với nhau ra sao
correlation_matrix = sales_df[['Order Quantity', 'Unit Price', 'Revenue', 'Profit']].corr()

2.	Khớp nối (Mapping) với các bảng vệ tinh:
o	Đối chiếu Customer Name Index sang bảng Customers.
o	Đối chiếu Delivery Region Index sang bảng Regions.
o	Đối chiếu Product Description Index sang bảng Products.
3.	Phát hiện dữ liệu khuyết liên kết: Giảng viên nhận thấy cột Warehouse Code trong bảng Sales Orders không có bất kỳ bảng (sheet) nào khác để giải nghĩa. Quyết định được đưa ra là sẽ tạm thời bỏ qua (skip) cột này trong việc xây dựng mô hình quan hệ.
4.	Liên kết gián tiếp: Bảng 2017 Budgets không nối trực tiếp với bảng Sales, mà nối với bảng Products thông qua cột Product Name.
5.	Xác định bản chất mối quan hệ (Cardinality): Đánh giá xem quan hệ giữa các bảng là 1-1 (One-to-One) hay 1-N (One-to-Many).
________________________________________
3. Kiến trúc & Quy trình phân tích (Mô hình Dữ liệu - ERD)
Dựa trên phân tích của giảng viên, bộ dữ liệu Regional Sales tuân theo kiến trúc lõi sau:
•	Bảng Trung tâm (Fact Table): Sales Orders
o	Primary Key (Khóa chính): Order Number
•	Các Mối quan hệ 1-N (Một - Nhiều):
o	Customers (Khách hàng) $\rightarrow$ Sales Orders: * Liên kết qua: Customer Index = Customer Name Index.
	Logic: 1 khách hàng có thể mua nhiều đơn hàng (1-to-N).
o	Regions (Khu vực) $\rightarrow$ Sales Orders:
	Liên kết qua: ID = Delivery Region Index.
	Logic: 1 khu vực địa lý có thể nhận giao nhiều đơn hàng (1-to-N).
o	Products (Sản phẩm) $\rightarrow$ Sales Orders:
	Liên kết qua: Index = Product Description Index.
	Logic: 1 sản phẩm có thể nằm trong nhiều đơn đặt hàng khác nhau (1-to-N).
•	Mối quan hệ mở rộng (Snowflake):
o	Products $\leftrightarrow$ 2017 Budgets: Hai bảng này được liên kết với nhau thông qua cột Product Name.

Để thực sự làm chủ cách tổ chức dữ liệu như giảng viên đang hướng dẫn, bạn cần nắm vững các khái niệm mở rộng sau:
•	Fact Table vs. Dimension Table (Bảng Sự kiện & Bảng Chiều):
o	Fact Table (Bảng Sự kiện): Trong dự án này là bảng Sales Orders. Nó chứa các con số có thể tính toán, cộng trừ nhân chia được (như Order Quantity, Unit Price, Line Total) và chứa các Khóa ngoại. Dữ liệu trong bảng này tăng lên mỗi ngày.
o	Dimension Table (Bảng Chiều): Là các bảng Customers, Regions, Products. Chúng chứa ngữ cảnh (Text) mô tả chi tiết cho Fact Table (Ví dụ: Ai mua? Mua ở đâu? Tên sản phẩm là gì?).
•	Star Schema (Lược đồ hình sao):
o	Mạng lưới ERD mà giảng viên vừa vẽ bằng lời mô tả chính là nền tảng của Star Schema. Đây là mô hình tối ưu nhất cho báo cáo Power BI. Bảng Sales nằm ở trung tâm (như lõi ngôi sao), các bảng Customers, Regions, Products nằm ở xung quanh (các cánh sao).
•	Auto-Detect Relationship trong Power BI:
o	Giảng viên có nhắc: "Power BI will automatically be able to see the relationship...". Khi bạn Import các file CSV này vào Power BI, AI của công cụ sẽ tự động tìm các cột có tên giống nhau hoặc dữ liệu giống nhau để nối lại. Tuy nhiên, AI thường xuyên nối sai. Việc bạn vẽ được ERD ra giấy/Miro board trước sẽ giúp bạn tự tay chỉnh sửa lại sơ đồ (Data Modeling) trong Power BI một cách chính xác tuyệt đối.

4. Phân tích các file code (Dựa trên phiên âm)
Dưới đây là phần code logic được tái hiện lại từ phiên âm của giảng viên, kèm theo các giải thích trực tiếp (comments) bằng tiếng Việt:
Python
# ==========================================
# KHỐI 1: IMPORT & ĐỌC DỮ LIỆU
# Mục đích: Đưa toàn bộ các sheet Excel vào bộ nhớ Python
# ==========================================
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Tham số sheet_name=None giúp đọc tất cả các sheet vào một Dictionary
sheets = pd.read_excel('Regional Sales Dataset.xlsx', sheet_name=None)

# Tách từng sheet thành các DataFrame riêng biệt
df_sales = sheets['Sales Orders']
df_customers = sheets['Customers']
df_products = sheets['Products']
df_regions = sheets['Regions']
df_state_regions = sheets['State Regions']
df_budgets = sheets['2017 Budgets']

# ==========================================
# KHỐI 2: SỬA LỖI HEADER CHO BẢNG STATE REGIONS
# Mục đích: Đưa dòng dữ liệu đầu tiên lên làm Header (Tiêu đề cột)
# ==========================================
# Lấy dòng đầu tiên (chứa tên cột bị sai vị trí) làm Header mới
new_header = df_state_regions.iloc[0] 
# Gán Header mới cho DataFrame
df_state_regions.columns = new_header 
# Xóa dòng đầu tiên đó đi và reset lại số thứ tự (index)
df_state_regions = df_state_regions[1:].reset_index(drop=True)

# ==========================================
# KHỐI 3: MERGING (GỘP BẢNG) DỰA TRÊN MÔ HÌNH ERD
# Mục đích: Nối các thông tin chi tiết vào bảng Sales
# ==========================================
# Dùng Left Join để giữ nguyên toàn bộ đơn hàng trong df_sales
# Nối bảng Khách Hàng
df = df_sales.merge(df_customers, how='left', left_on='Customer Name Index', right_on='Customer Index')

# Nối bảng Sản Phẩm
df = df.merge(df_products, how='left', left_on='Product Description Index', right_on='Index')

# Nối bảng Khu Vực (Regions)
df = df.merge(df_regions, how='left', left_on='Delivery Region Index', right_on='ID')

# Nối bảng State Regions (Lấy cột State Code làm cầu nối)
df = df.merge(df_state_regions[['State Code', 'Region']], how='left', on='State Code')

# Nối bảng Budgets 2017 (Nối trực tiếp qua tên sản phẩm)
df = df.merge(df_budgets, how='left', on='Product Name')

# ==========================================
# KHỐI 4: LỌC CỘT VÀ CHUẨN HÓA (DATA REDUCTION)
# Mục đích: Xóa các cột rác, đổi tên cột chuẩn mực
# ==========================================
# Danh sách các cột cần giữ lại (Loại bỏ Time zone, Population, Currency...)
cols_to_keep = [
    'OrderNumber', 'OrderDate', 'Customer Names', 'Channel', 'Product Name', 
    'Order Quantity', 'Unit Price', 'Line Total', 'Total Unit Cost', 
    'State Code', 'County', 'State', 'Region', 'latitude', 'longitude', '2017 Budgets'
]
# Ghi đè DataFrame chỉ với các cột đã chọn
df = df[cols_to_keep]

# Đổi tên cột cho dễ thao tác, chuyển Line Total thành Revenue (Doanh thu)
df.rename(columns={
    'OrderNumber': 'order_number',
    'OrderDate': 'order_date',
    'Customer Names': 'customer_name',
    'Product Name': 'product_name',
    'Line Total': 'revenue',           # <-- Feature Engineering cơ bản
    '2017 Budgets': 'budget'
}, inplace=True)

# ==========================================
# KHỐI 5: XỬ LÝ LỖI LOGIC THỜI GIAN (BUDGETS)
# Mục đích: Ngân sách năm 2017 không được áp dụng cho các năm khác
# ==========================================
# Ép kiểu dữ liệu thời gian cho Order Date
df['order_date'] = pd.to_datetime(df['order_date'])

# Logic: Tìm các dòng có năm KHÁC 2017, cập nhật cột 'budget' của các dòng đó thành giá trị rỗng (NaN)
df.loc[df['order_date'].dt.year != 2017, 'budget'] = pd.NA

5. Kiến thức liên quan & mở rộng
Trong buổi học này, giảng viên đã áp dụng một số kinh nghiệm thực chiến cực kỳ quan trọng mà một Data Analyst cần có:
•	Tại sao luôn dùng Left Join thay vì Inner Join?
o	Trong kinh doanh, bảng Sales Orders là "sự thật" (Fact). Việc dùng Left Join đảm bảo rằng không có bất kỳ một hóa đơn bán hàng nào bị mất đi, ngay cả khi thông tin khách hàng hay thông tin khu vực đó bị thiếu hoặc bị lỗi trong hệ thống dữ liệu.
•	Tư duy loại bỏ biến (Feature Dropping): * Giảng viên đã bỏ cột Population (Dân số) và Area Code (Mã vùng) vì thấy không liên quan đến bài toán Doanh thu. Tuy nhiên, giảng viên có mở rộng rằng: Nếu bài toán yêu cầu phân tích "Tỷ lệ thâm nhập thị trường" (Market Penetration = Doanh thu / Dân số), thì cột Population lại rất quan trọng. Điều này minh chứng cho việc: Việc giữ hay bỏ dữ liệu phụ thuộc hoàn toàn vào Business Objective (Mục tiêu kinh doanh).
•	Xử lý Temporal Mismatch (Lệch pha thời gian):
o	Hành động "xóa trắng" cột Budget của các năm 2014, 2015, 2016, 2018 là một tư duy xử lý dữ liệu xuất sắc. Nếu không làm bước này, khi đưa dữ liệu lên Power BI, tổng ngân sách (Total Budget) của 5 năm sẽ bị tính sai (bị nhân lên gấp 5 lần), dẫn đến việc so sánh KPI Thực tế (Actual) vs Kế hoạch (Budget) hoàn toàn sai lệch. Giá trị pd.NA (Null) sẽ được Power BI hoặc hàm Sum() của Pandas tự động bỏ qua khi tính toán.
